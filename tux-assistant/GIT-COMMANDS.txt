# ============================================================================
# Tux Assistant - Git Reorganization Commands
# Copy and paste each section one at a time
# ============================================================================

# ============================================================================
# STEP 1: Navigate to your repo and create the dev branch
# ============================================================================

cd ~/path/to/tux-assistant
git checkout main
git pull origin main
git checkout -b dev
git push -u origin dev
git checkout main


# ============================================================================
# STEP 2: After copying new files from Claude, commit dev branch
# (Run this AFTER you've extracted the new zip and replaced files)
# ============================================================================

git checkout dev
git add .
git commit -m "v0.9.0 - Prepare for public release"
git push origin dev


# ============================================================================
# STEP 3: Prepare main branch for public release
# (This removes source code, keeps only release files)
# ============================================================================

git checkout main

# Remove source code (will only exist on dev branch)
rm -rf tux/
rm -rf scripts/
rm -rf docs/
rm -f tux-assistant.py
rm -f tux-helper
rm -f install.sh
rm -f TODO.md
rm -f .gitignore

# Rename public README
mv README-public.md README.md

# Create releases directory and move .run file
mkdir -p releases
mv dist/Tux-Assistant-v0.9.0.run releases/

# Clean up
rm -rf dist/
rm -rf build/


# ============================================================================
# STEP 4: Commit and push main (public release)
# ============================================================================

git add .
git commit -m "v0.9.0 - First public release"
git push origin main


# ============================================================================
# STEP 5: Create GitHub Release (do this on GitHub website)
# ============================================================================
# 1. Go to: https://github.com/dorrelkc/tux-assistant/releases
# 2. Click "Create a new release"
# 3. Tag version: v0.9.0
# 4. Release title: Tux Assistant v0.9.0 - First Public Release
# 5. Upload: releases/Tux-Assistant-v0.9.0.run
# 6. Publish release
